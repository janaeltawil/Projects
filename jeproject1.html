data$Track_pop_Categorical <- cut(data$track_pop,

                                  labels = c("Unpopular", "Somewhat Popular", "Moderately Popular", "Popular", "Very Popular"),
                                  include.lowest = TRUE)





# Creating a data frame with the selected features

target_data <- data[, c("Track_pop_Categorical", "Tempo_Categorical", "Valence_Categorical", "Acousticness_Categorical", "Speechiness_Categorical", "Liveness_Categorical", "Instrumentalness_Categorical", "key", "mode")]

# Splitting the data and setting seed for reproducibility
set.seed(123) 

train_index <- sample(1:nrow(target_data), 0.7 * nrow(target_data)) 
train_data <- target_data[train_index, ]
test_data <- target_data[-train_index, ]


## 1. Decision Tree Model
model <- rpart(Track_pop_Categorical ~ Tempo_Categorical + Valence_Categorical + Acousticness_Categorical + Speechiness_Categorical + Liveness_Categorical + Instrumentalness_Categorical + key + mode, data = train_data, method = "class")

# Decision tree
rpart.plot(model, type = 4, extra = 104, main = "Decision Tree")

# a. Prediction and Confusion Matrix
pred <- predict(model, newdata = test_data, type = "class")
confusion_matrix <- table(Predicted = pred, Actual = test_data$Track_pop_Categorical)
accuracy <- mean(pred == test_data$Track_pop_Categorical)

print(confusion_matrix)
